üíª Codice Sorgente Commentato

// OWNERSHIP
fn main() {
    let s1 = String::from("Rust");
    let s2 = s1;  // Move
    println!("{}", s2);
}

// BORROWING
fn calcola_lunghezza(s: &String) -> usize {
    s.len()
}

fn main() {
    let s3 = String::from("Ownership");
    let len = calcola_lunghezza(&s3);
    println!("'{}' ha lunghezza {}", s3, len);
}

// MUTABLE BORROWING
fn modifica(s: &mut String) {
    s.push_str(" mondo!");
}

fn main() {
    let mut testo = String::from("Ciao");
    modifica(&mut testo);
    println!("{}", testo);
}
