üìò Introduzione Teorica

Una documentazione professionale √® essenziale per:

  • README.md: Prima impressione, setup, esempi
  • Cargo.toml metadata: Descrizione, keywords, categorie
  • Rustdoc: Commenti di documentazione inline
  • CONTRIBUTING.md: Come contribuire al progetto

// ========================================
// src/lib.rs - Documentazione modulo
// ========================================

//! # Text Editor Library
//!
//! Una libreria per editor di testo con GTK.
//!
//! ## Example
//! ```
//! use text_editor::Editor;
//! let editor = Editor::new();
//! ```

/// Rappresenta un editor di testo
pub struct Editor {
    /// Contenuto attuale
    content: String,
}

impl Editor {
    /// Crea un nuovo editor vuoto
    pub fn new() -> Self {
        Editor {
            content: String::new(),
        }
    }

    /// Carica file da percorso
    ///
    /// # Arguments
    /// * `path` - Percorso del file
    ///
    /// # Returns
    /// * `Result<(), std::io::Error>` - OK o errore I/O
    ///
    /// # Errors
    /// Ritorna errore se file non esiste o non leggibile
    pub fn load_file(&mut self, path: &str) -> std::io::Result<()> {
        self.content = std::fs::read_to_string(path)?;
        Ok(())
    }

    /// Ritorna il contenuto attuale
    pub fn get_content(&self) -> &str {
        &self.content
    }

    /// Inserisce testo alla posizione
    ///
    /// # Panics
    /// Panics se posizione > lunghezza contenuto
    pub fn insert(&mut self, pos: usize, text: &str) {
        self.content.insert_str(pos, text);
    }
}
